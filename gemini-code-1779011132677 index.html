import pygame import random import sys # Initialize Pygame pygame.init() # --- CONSTANTS & GAME CONFIG --- SCREEN_WIDTH = 500 SCREEN_HEIGHT = 600 FPS = 60 # Colors (RGB) BLACK = (0, 0, 0) WHITE = (255, 255, 255) GRAY = (50, 50, 50) YELLOW = (255, 255, 0) RED = (220, 20, 60) BLUE = (30, 144, 255) GREEN = (50, 205, 50) # Create the game window screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) # ======================================================= # এখানে আপনার পছন্দমতো গেমের টাইটেল বা নাম সেট করা হয়েছে: pygame.display.set_caption("Amit's Car Game") # ======================================================= clock = pygame.time.Clock() # Fonts font_score = pygame.font.SysFont("Arial", 24, bold=True) font_game_over = pygame.font.SysFont("Arial", 50, bold=True) font_restart = pygame.font.SysFont("Arial", 20) # --- GAME OBJECT CLASSES --- class PlayerCar: """Class representing the user-controlled car.""" def __init__(self): self.width = 40 self.height = 70 self.x = (SCREEN_WIDTH // 2) - (self.width // 2) self.y = SCREEN_HEIGHT - self.height - 30 self.speed = 6 self.min_x = 80 self.max_x = SCREEN_WIDTH - 80 - self.width def handle_keys(self): keys = pygame.key.get_pressed() if keys[pygame.K_LEFT]: self.x -= self.speed if keys[pygame.K_RIGHT]: self.x += self.speed if self.x < self.min_x: self.x = self.min_x elif self.x > self.max_x: self.x = self.max_x def draw(self, surface): pygame.draw.rect(surface, BLUE, (self.x, self.y, self.width, self.height), border_radius=5) pygame.draw.rect(surface, WHITE, (self.x + 5, self.y + 15, self.width - 10, 20), border_radius=3) pygame.draw.rect(surface, BLACK, (self.x - 5, self.y + 10, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x + self.width, self.y + 10, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x - 5, self.y + 45, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x + self.width, self.y + 45, 5, 15)) def get_rect(self): return pygame.Rect(self.x, self.y, self.width, self.height) class ObstacleCar: """Class representing oncoming traffic obstacles.""" def __init__(self, speed): self.width = 40 self.height = 70 self.x = random.randint(80, SCREEN_WIDTH - 80 - self.width) self.y = random.randint(-300, -100) self.speed = speed self.color = random.choice([RED, GREEN, YELLOW]) def update(self): self.y += self.speed def draw(self, surface): pygame.draw.rect(surface, self.color, (self.x, self.y, self.width, self.height), border_radius=5) pygame.draw.rect(surface, BLACK, (self.x + 5, self.y + self.height - 35, self.width - 10, 20), border_radius=3) pygame.draw.rect(surface, BLACK, (self.x - 5, self.y + 10, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x + self.width, self.y + 10, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x - 5, self.y + 45, 5, 15)) pygame.draw.rect(surface, BLACK, (self.x + self.width, self.y + 45, 5, 15)) def is_off_screen(self): return self.y > SCREEN_HEIGHT def get_rect(self): return pygame.Rect(self.x, self.y, self.width, self.height) # --- ENVIRONMENT GENERATION --- def draw_background(surface, lane_offset): surface.fill((34, 139, 34)) road_left = 70 road_width = SCREEN_WIDTH - 140 pygame.draw.rect(surface, GRAY, (road_left, 0, road_width, SCREEN_HEIGHT)) pygame.draw.line(surface, WHITE, (road_left, 0), (road_left, SCREEN_HEIGHT), 4) pygame.draw.line(surface, WHITE, (road_left + road_width, 0), (road_left + road_width, SCREEN_HEIGHT), 4) lane_x = SCREEN_WIDTH // 2 dash_length = 30 gap_length = 20 y = -dash_length + lane_offset while y < SCREEN_HEIGHT: pygame.draw.line(surface, YELLOW, (lane_x, y), (lane_x, y + dash_length), 3) y += dash_length + gap_length # --- MAIN GAME LOOP CONTROLLER --- def main(): player = PlayerCar() obstacles = [] base_obstacle_speed = 4 max_obstacles = 3 score = 0 lane_offset = 0 game_over = False running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if game_over and event.key == pygame.K_r: main() return if not game_over: player.handle_keys() lane_offset += base_obstacle_speed if lane_offset >= 50: lane_offset = 0 current_speed = base_obstacle_speed + (score // 10) if len(obstacles) < max_obstacles: if len(obstacles) == 0 or obstacles[-1].y > 150: obstacles.append(ObstacleCar(current_speed)) for obs in obstacles: obs.update() for obs in obstacles[:]: if obs.is_off_screen(): obstacles.remove(obs) score += 1 player_rect = player.get_rect() for obs in obstacles: if player_rect.colliderect(obs.get_rect()): game_over = True draw_background(screen, lane_offset) player.draw(screen) for obs in obstacles: obs.draw(screen) score_text = font_score.render(f"Score: {score}", True, WHITE) screen.blit(score_text, (15, 15)) if game_over: overlay = pygame.Surface((SCREEN_WIDTH, SCREEN_HEIGHT), pygame.SRCALPHA) overlay.fill((0, 0, 0, 150)) screen.blit(overlay, (0, 0)) go_text = font_game_over.render("GAME OVER", True, RED) go_rect = go_text.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2 - 40)) screen.blit(go_text, go_rect) final_score_text = font_score.render(f"Final Score: {score}", True, WHITE) fs_rect = final_score_text.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2 + 15)) screen.blit(final_score_text, fs_rect) restart_text = font_restart.render("Press 'R' to Restart Game", True, YELLOW) r_rect = restart_text.get_rect(center=(SCREEN_WIDTH // 2, SCREEN_HEIGHT // 2 + 60)) screen.blit(restart_text, r_rect) pygame.display.flip() clock.tick(FPS) if __name__ == "__main__": main()
